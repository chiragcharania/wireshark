1.	planetlab-1.sjtu.edu.cn (China):

ssh -v -l albany_ccn8 -i ~/.ssh/id_rsa planetlab-1.sjtu.edu.cn

mkdir cc

cd cc

vim my_script.sh
#!/bin/bash
declare -a hostName=("pl1.rcc.uottawa.ca" "planetlab2.cs.purdue.edu" "planetlab3.cs.uoregon.edu");
while [ 1 ]; do
date
for i in "${hostName[@]}"
do
printf "\n"
printf "Ping from planetlab-1.sjtu.edu.cn:"
ping -c 20 $i
printf "\n"
printf "Traceroute from planetlab-1.sjtu.edu.cn:"
traceroute $i
done
sleep 1200
done

chmod +x my_script.sh

setsid ./my_script.sh > node1_1_log.txt
